import itertools

#       
edges = [(1,2,3), (1,3,5), (1,4,7), (2,3,6), (2,4,8), (3,4,9)]

#        
graph = {}
for u, v, w in edges:
    graph[(u, v)] = w
    graph[(v, u)] = w  #   

#   
nodes = [1, 2, 3, 4]

min_path = None
min_cost = float('inf')

#     1
start = 1
#      
for perm in itertools.permutations([n for n in nodes if n != start]):
    path = [start] + list(perm) + [start]  #  
    cost = 0
    for i in range(len(path) - 1):
        cost += graph[(path[i], path[i+1])]
    if cost < min_cost:
        min_cost = cost
        min_path = path

#  
print(" :", min_path)
print(" :", min_cost)
